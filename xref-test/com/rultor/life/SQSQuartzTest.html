54  /**
55   * Test case for {@link SQSQuartz}.
56   * @author Yegor Bugayenko (yegor@tpc2.com)
57   * @version $Id$
58   * @checkstyle ClassDataAbstractionCoupling (500 lines)
59   */
60  public final class SQSQuartzTest {
61  
62      /**
63       * SQSQuartz can publish works into queue.
64       * @throws Exception If some problem inside
65       * @checkstyle ExecutableStatementCount (200 lines)
66       */
67      @Test
68      public void publishesWorksIntoQueue() throws Exception {
69          final Users users = Mockito.mock(Users.class);
70          final URN urn = new URN("urn:github:1");
71          Mockito.doReturn(new HashSet<URN>(Arrays.asList(urn)))
72              .when(users).everybody();
73          final User user = Mockito.mock(User.class);
74          Mockito.doReturn(urn).when(user).urn();
75          Mockito.doReturn(user).when(users).get(urn);
76          final String name = "unit-name";
77          Mockito.doReturn(new HashSet<String>(Arrays.asList(name)))
78              .when(user).units();
79          final Unit unit = Mockito.mock(Unit.class);
80          Mockito.doReturn(unit).when(user).get(name);
81          Mockito.doReturn(new Spec.Simple()).when(unit).spec();
82          final Queue queue = Mockito.mock(Queue.class);
83          final SQSClient client = Mockito.mock(SQSClient.class);
84          final AmazonSQS aws = Mockito.mock(AmazonSQS.class);
85          final Time time = new Time("2013-07-21T13:36Z");
86          final String handle = "some-random-text";
87          Mockito.doReturn(aws).when(client).get();
88          Mockito.doReturn(
89              new ReceiveMessageResult().withMessages(
90                  new Message()
91                      .withBody(time.toString())
92                      .withReceiptHandle(handle)
93              )
94          ).when(aws).receiveMessage(Mockito.any(ReceiveMessageRequest.class));
95          final SQSQuartz quartz = new SQSQuartz(users, queue, client);
96          quartz.run();
97          Mockito.verify(aws).receiveMessage(
98              Mockito.any(ReceiveMessageRequest.class)
99          );
100         Mockito.verify(aws).sendMessage(
101             Mockito.argThat(
102                 Matchers.<SendMessageRequest>hasProperty(
103                     "messageBody",
104                     Matchers.equalTo("2013-07-21T13:37Z")
105                 )
106             )
107         );
108         Mockito.verify(aws).deleteMessage(
109             Mockito.argThat(
110                 Matchers.<DeleteMessageRequest>hasProperty(
111                     "receiptHandle",
112                     Matchers.equalTo(handle)
113                 )
114             )
115         );
116         Mockito.verify(queue).push(
117             Mockito.argThat(
118                 new CustomMatcher<Work>("expected work") {
119                     @Override
120                     public boolean matches(final Object work) {
121                         return Work.class.cast(work).started().equals(time);
122                     }
123                 }
124             )
125         );
126     }
