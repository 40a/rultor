57      /**
58       * ClasspathRepo can make a spec.
59       * @throws Exception If some problem inside
60       */
61      @Test
62      public void makesSpecFromText() throws Exception {
63          final Repo repo = new ClasspathRepo();
64          final String[] texts = new String[] {
65              "com.rultor.repo.ClasspathRepoTest$Foo(55)",
66              "java.lang.String( 'te   \n st' )",
67              "java.lang.Integer ( 123 )",
68              "java.lang.Long(-44L)",
69              "foo.SomeClass(1, FALSE, TRUE, 8L, 'test')",
70              "java.lang.Double(-44.66)",
71              "com.rultor.repo.ClasspathRepoTest$Foo (55\n)",
72              "com.first(com.second(com.third(), com.forth()))",
73              "[com.second(com.third(), com.forth()), 4]",
74              "{\"test\": com.second(com.third(), com.forth()), \"c\": 77}",
75              "\"\"\"\nsome\nunformatted text \u20ac\u0433\n\"\"\"",
76          };
77          final User user = Mockito.mock(User.class);
78          final URN urn = new URN("urn:facebook:1");
79          Mockito.doReturn(urn).when(user).urn();
80          for (final String text : texts) {
81              MatcherAssert.assertThat(
82                  repo.make(user, new Spec.Simple(text)).asText(),
83                  Matchers.equalTo(
84                      repo.make(user, new Spec.Simple(text)).asText()
85                  )
86              );
87          }
88      }

99              "java.lang.Boolean(TRUE)",
100             "\"\"\"\nsome\u20actext\n\t\"\"\"",
101         };
102         final User user = Mockito.mock(User.class);
103         final URN urn = new URN("urn:facebook:2");
104         Mockito.doReturn(urn).when(user).urn();
105         for (final String text : texts) {
106             final Spec spec = new Spec.Simple(text);
107             final Object obj = repo.make(user, spec);
108             MatcherAssert.assertThat(obj, Matchers.notNullValue());
109         }
110     }

133     /**
134      * ClasspathRepo can make an instance.
135      * @throws Exception If some problem inside
136      */
137     @Test
138     public void makesInstanceFromSpec() throws Exception {
139         final Repo repo = new ClasspathRepo();
140         final Spec multiply = new Spec.Simple(
141             // @checkstyle StringLiteralsConcatenation (4 lines)
142             "com.rultor.repo.ClasspathRepoTest$Plus(  "
143             + "  com.rultor.repo.ClasspathRepoTest$Const(${0:test}),"
144             + "  com.rultor.repo.ClasspathRepoTest$Const(${0:test} )"
145             + ") "
146         );
147         final Spec spec = new Spec.Simple(
148             // @checkstyle StringLiteralsConcatenation (8 lines)
149             "com.rultor.repo.ClasspathRepoTest$Plus("
150             + "  com.rultor.repo.ClasspathRepoTest$Plus("
151             + "    com.rultor.repo.ClasspathRepoTest$Const(5), "
152             + "    multiply-by-two(-2)"
153             + "  ),"
154             + "  com.rultor.repo.ClasspathRepoTest$Const(${0:test again})"
155             + ")"
156         );
157         final Rule rule = Mockito.mock(Rule.class);
158         Mockito.doReturn(multiply).when(rule).spec();
159         final String name = "multiply-by-two";
160         final User user = Mockito.mock(User.class);
161         final Rules rules = Mockito.mock(Rules.class);
162         Mockito.doReturn(rules).when(user).rules();
163         Mockito.doReturn(rule).when(rules).get(name);
164         final URN urn = new URN("urn:facebook:77");
165         Mockito.doReturn(urn).when(user).urn();
166         final Users users = Mockito.mock(Users.class);
167         Mockito.doReturn(user).when(users).get(urn);
168         final ClasspathRepoTest.Atom atom = ClasspathRepoTest.Atom.class.cast(
169             repo.make(user, spec).instantiate(
170                 users,
171                 new Arguments(
172                     new Coordinates.None(), new Wallet.Empty(),
173                     Arrays.<Object>asList(-2)
174                 )
175             )
176         );
177         MatcherAssert.assertThat(atom.calc(), Matchers.equalTo(-1));
178     }

180     /**
181      * Atom.
182      */
183     @Immutable
184     public interface Atom {
185         /**
186          * Get its value.
187          * @return Result
188          */
189         int calc();
190     }
191 
192     /**
193      * Just number.
194      */
195     @Immutable
196     public static final class Const implements ClasspathRepoTest.Atom {
197         /**
198          * Number.
199          */
200         private final transient int value;
201         /**
202          * Public ctor.
203          * @param val The number
204          */
205         public Const(final int val) {
206             this.value = val;
207         }
208         @Override
209         public int calc() {
210             return this.value;
211         }
212     }
213 
214     /**
215      * Number plus number.
216      */
217     @Immutable
218     public static final class Plus implements ClasspathRepoTest.Atom {
219         /**
220          * Left number.
221          */
222         private final transient ClasspathRepoTest.Atom left;
223         /**
224          * Right number.
225          */
226         private final transient ClasspathRepoTest.Atom right;
227         /**
228          * Public ctor.
229          * @param first The number
230          * @param second The number
231          */
232         public Plus(final ClasspathRepoTest.Atom first,
233             final ClasspathRepoTest.Atom second) {
234             this.left = first;
235             this.right = second;
236         }
237         @Override
238         public int calc() {
239             return this.left.calc() + this.right.calc();
240         }
241     }
