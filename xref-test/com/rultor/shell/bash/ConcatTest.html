56      /**
57       * Concat can run complex script with bash.
58       * @throws Exception If some problem inside
59       */
60      @Test
61      public void runsIncrementalBashScript() throws Exception {
62          final ByteArrayOutputStream stdout = new ByteArrayOutputStream();
63          final File dir = Files.createTempDir();
64          FileUtils.write(new File(dir, "a.txt"), "first\nsecond");
65          final ImmutableMap<String, String> args =
66              new ImmutableMap.Builder<String, String>()
67                  .put("file", "file-name.txt")
68                  .build();
69          final int code = new Bash(
70              new Permanent(new ShellMocker.Bash(dir)),
71              new Concat(
72                  Arrays.asList(
73                      "MSG='$A'; echo $(date) $A; sleep 1; pwd;",
74                      "find . -name \"a.txt\" | grep txt | wc -l;",
75                      "mkdir -p foo; cd foo; touch ${file}; pwd",
76                      "pwd; if [ ! -f ${file.toString()} ]; then exit 1; fi",
77                      "echo -e \"\\x1b\\x09\\x9B\" >&2; /usr/--bad; /usr/--again"
78                  )
79              ).object()
80          ).exec(args, stdout);
81          MatcherAssert.assertThat(code, Matchers.not(Matchers.equalTo(0)));
82          MatcherAssert.assertThat(
83              XhtmlMatchers.xhtml(
84                  new Snapshot(
85                      new ByteArrayInputStream(stdout.toByteArray())
86                  ).xml()
87              ),
88              XhtmlMatchers.hasXPaths(
89                  "/snapshot/steps/step",
90                  // @checkstyle LineLength (5 lines)
91                  "//step[summary=\"MSG='$A'; echo $(date) $A; sleep 1; pwd;\"]/start",
92                  "//step[contains(summary, '/usr/--bad; /usr/--again')]/exception",
93                  "//step/exception[contains(stacktrace,'/usr/--bad: No such file or directory')]",
94                  "//steps[count(step[level='INFO']) = 4]",
95                  "//steps[count(step[level='SEVERE']) = 1]",
96                  "//steps[count(step[start]) = 5]",
97                  "//steps[count(step[finish]) = 5]",
98                  "//steps[count(step[duration = '']) = 0]"
99              )
100         );
101     }

107     @Test
108     public void escapesMarkdownInCommandSummary() throws Exception {
109         final ByteArrayOutputStream stdout = new ByteArrayOutputStream();
110         final File dir = Files.createTempDir();
111         final int code = new Bash(
112             new Permanent(new ShellMocker.Bash(dir)),
113             new Concat(Arrays.asList("echo \"_*\" `date`")).object()

115         MatcherAssert.assertThat(code, Matchers.equalTo(0));

120                 ).xml()

192     /**
193      * Concat can output stderr and stdout correctly.
194      * @throws Exception If some problem inside
195      */
196     @Test
197     public void correctlyPrintsStderrAndStdout() throws Exception {
198         final ByteArrayOutputStream stdout = new ByteArrayOutputStream();
199         final File dir = Files.createTempDir();
200         new Bash(
201             new Permanent(new ShellMocker.Bash(dir)),
202             new Concat(
203                 Arrays.asList(
204                     "echo -e 'one\\ntwo' >&2; echo -e 'foo-1\\nfoo-2'"
205                 )
206             ).object()
207         ).exec(new ImmutableMap.Builder<String, String>().build(), stdout);
208         MatcherAssert.assertThat(
209             new String(stdout.toByteArray(), CharEncoding.UTF_8),
210             Matchers.allOf(
211                 Matchers.containsString("one\ntwo"),
212                 Matchers.containsString("foo-1\nfoo-2")
213             )
214         );
