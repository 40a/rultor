49      /**
50       * PostSnapshot can post-process a snapshot with ETA.
51       * @throws Exception If some problem inside
52       */
53      @Test
54      public void postProcessesSnapshot() throws Exception {
55          MatcherAssert.assertThat(
56              this.xml(
57                  new Directives()
58                      .xpath("/snapshot")
59                      .add("start")
60                      .set("2012-08-23T13:00:00Z")
61                      .up()
62                      .add("eta")
63                      .set("2012-08-23T14:00:00Z")
64                      .up()
65                      .add("updated")
66                      .set("2012-08-23T13:30:00Z")
67                      .up()
68                      .add("steps")
69                      .add("step")
70                      .attr("id", "7")
71                      .add("start")
72                      .set("2012-08-23T13:06:00Z")
73                      .up()
74                      .add("finish")
75                      .set("2012-08-23T13:15:00Z")
76                      .toString()
77              ),
78              XhtmlMatchers.hasXPaths(
79                  "/snapshot/updated[@at='0.5']",
80                  "/snapshot/steps/step[@id=7]/start[@at='0.1']",
81                  "/snapshot/steps/step[@id=7]/finish[@at='0.25']"
82              )
83          );
84      }

180     /**
181      * Make XML out of directives.
182      * @param xembly Xembly
183      * @return XML source
184      * @throws Exception If fails
185      */
186     private Source xml(final String xembly) throws Exception {
187         return XhtmlMatchers.xhtml(
188             new XSLT(
189                 new Snapshot(xembly),
190                 this.getClass().getResourceAsStream("post.xsl")
191             ).dom()
192         );
193     }
