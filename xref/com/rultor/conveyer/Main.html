44  /**
45   * Main entry point to the JAR.
46   *
47   * @author Yegor Bugayenko (yegor@tpc2.com)
48   * @version $Id$
49   * @since 1.0
50   * @checkstyle ClassDataAbstractionCoupling (500 lines)
51   * @checkstyle MultipleStringLiterals (500 lines)
52   */
53  @ToString
54  @EqualsAndHashCode
55  @SuppressWarnings({ "PMD.ExcessiveImports", "PMD.AvoidDuplicateLiterals" })
56  @Loggable(Loggable.DEBUG)
57  public final class Main {
58  
59      /**
60       * Random.
61       */
62      private static final Random RND = new SecureRandom();
63  
64      /**
65       * It's a utility class.
66       */
67      private Main() {
68          // intentionally empty
69      }
70  
71      /**
72       * Entrance.
73       * @param args Optional arguments
74       * @throws Exception If something is wrong
75       */
76      @Loggable(value = Loggable.DEBUG, limit = Integer.MAX_VALUE)
77      public static void main(final String[] args) throws Exception {
78          final OptionParser parser = Main.parser();
79          final OptionSet options = parser.parse(args);
80          if (options.has("help")) {
81              parser.printHelpOn(Logger.stream(Level.INFO, Main.class));
82          } else {
83              final SimpleConveyer conveyer =
84                  new ConveyerBuilder(options).build();
85              final long start = System.currentTimeMillis();
86              conveyer.start();
87              while (Main.alive(start, options)) {
88                  TimeUnit.SECONDS.sleep(1);
89              }
90              conveyer.close();
91          }
92      }
93  
94      /**
95       * Are we still need to be alive?
96       * @param start When scheduled
97       * @param options Command line options
98       * @return TRUE if we should still be alive
99       * @throws Exception If fails
100      */
101     private static boolean alive(final long start, final OptionSet options)
102         throws Exception {
103         final boolean alive;
104         if (options.has("lifetime")) {
105             final long lifetime = Long.parseLong(
106                 options.valueOf("lifetime").toString()
107             );
108             alive = System.currentTimeMillis() - start < lifetime;
109         } else {
110             if (new Version().same()) {
111                 TimeUnit.SECONDS.sleep(Main.RND.nextInt(Tv.HUNDRED));
112                 alive = true;
113             } else {
114                 alive = false;
115             }
116         }
117         return alive;
118     }
119 
120     /**
121      * Build a parser.
122      * @return Parser
123      */
124     private static OptionParser parser() {
125         final OptionParser parser = new OptionParser();
126         parser.accepts("help", "Show detailed instructions").forHelp();
127         parser.accepts("threads", "In how many threads to run")
128             .withRequiredArg().defaultsTo("5").ofType(String.class);
129         parser.accepts("spec", "Text file with work specification")
130             .withRequiredArg().ofType(String.class);
131         parser.accepts("lifetime", "Maximum lifetime of the daemon, in millis")
132             .withRequiredArg().ofType(String.class)
133             .defaultsTo(Long.toString(TimeUnit.DAYS.toMillis(Tv.FIVE)));
134         parser.accepts("dynamo-key", "Amazon DynamoDB access key")
135             .withRequiredArg().ofType(String.class);
136         parser.accepts("dynamo-secret", "Amazon DynamoDB secret key")
137             .withRequiredArg().ofType(String.class);
138         parser.accepts("dynamo-prefix", "Amazon DynamoDB table name prefix")
139             .withRequiredArg().ofType(String.class);
140         parser.accepts("sqs-key", "Amazon SQS access key")
141             .withRequiredArg().ofType(String.class);
142         parser.accepts("sqs-secret", "Amazon SQS secret key")
143             .withRequiredArg().ofType(String.class);
144         parser.accepts("sqs-url", "Amazon SQS URL")
145             .withRequiredArg().ofType(String.class);
146         parser.accepts("sqs-wallet-url", "Amazon SQS URL for wallets")
147             .withRequiredArg().ofType(String.class);
148         parser.accepts("pgsql-url", "PostgreSQL JDBC URL")
149             .withRequiredArg().ofType(String.class);
150         parser.accepts("pgsql-password", "PostgreSQL password")
151             .withRequiredArg().ofType(String.class);
152         return parser;
153     }
