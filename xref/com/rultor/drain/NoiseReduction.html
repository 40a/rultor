56  /**
57   * Noise reduction.
58   *
59   * @author Yegor Bugayenko (yegor@tpc2.com)
60   * @version $Id$
61   * @since 1.0
62   * @checkstyle ClassDataAbstractionCoupling (500 lines)
63   */
64  @Immutable
65  @ToString
66  @EqualsAndHashCode(of = { "work", "pattern", "visible", "dirty", "clean" })
67  @Loggable(Loggable.DEBUG)
68  public final class NoiseReduction implements Drain {
69  
70      /**
71       * Static set of clean drains.
72       */
73      private static final Set<Drain> GOOD = new CopyOnWriteArraySet<Drain>();
74  
75      /**
76       * Static set of dirty drains.
77       */
78      private static final Set<Drain> BAD = new CopyOnWriteArraySet<Drain>();
79  
80      /**
81       * Coordinates we're in.
82       */
83      private final transient Coordinates work;
84  
85      /**
86       * Regular expression pattern to match.
87       */
88      private final transient String pattern;
89  
90      /**
91       * How many dirty pulses should be visible.
92       */
93      private final transient int visible;
94  
95      /**
96       * Dirty pulses.
97       */
98      private final transient Drain dirty;
99  
100     /**
101      * Clean pulses.
102      */
103     private final transient Drain clean;
104 
105     /**
106      * Public ctor.
107      * @param wrk Coordinates we're in
108      * @param ptn Regular
109      * @param vsbl How many items should be visible from dirty drain
110      * @param drt Dirty drain
111      * @param cln Clean drain
112      * @checkstyle ParameterNumber (10 lines)
113      */
114     public NoiseReduction(
115         @NotNull(message = "work can't be NULL") final Coordinates wrk,
116         @NotNull(message = "pattern can't be NULL") final String ptn,
117         final int vsbl,
118         @NotNull(message = "dirty drain can't be NULL") final Drain drt,
119         @NotNull(message = "clean drain can't be NULL") final Drain cln) {
120         this.work = wrk;
121         this.pattern = ptn;
122         this.visible = vsbl;
123         this.dirty = drt;
124         this.clean = cln;
125     }
126 
127     @Override
128     public Pageable<Time, Time> pulses() throws IOException {
129         // @checkstyle AnonInnerLength (50 lines)
130         return new Pageable<Time, Time>() {
131             @Override
132             public Pageable<Time, Time> tail(final Time head)
133                 throws IOException {
134                 return NoiseReduction.this.clean.pulses().tail(head);
135             }
136             @Override
137             public Iterator<Time> iterator() {
138                 try {
139                     return new NoiseReduction.Distinct<Time>(
140                         Iterables.concat(
141                             Iterables.limit(
142                                 NoiseReduction.this.dirty.pulses(),
143                                 NoiseReduction.this.visible
144                             ),
145                             NoiseReduction.this.clean.pulses()
146                         ).iterator()
147                     );
148                 } catch (final IOException ex) {
149                     throw new IllegalStateException(ex);
150                 }
151             }
152         };
153     }
154 
155     @Override
156     public void append(final Iterable<String> lines)
157         throws IOException {
158         final boolean matched;
159         synchronized (NoiseReduction.GOOD) {
160             matched = !NoiseReduction.GOOD.contains(this.dirty)
161                 && Iterables.any(
162                     lines,
163                     new Predicate<String>() {
164                         @Override
165                         public boolean apply(final String input) {
166                             return input.matches(NoiseReduction.this.pattern);
167                         }
168                     }
169                 );
170             if (matched) {
171                 NoiseReduction.GOOD.add(this.dirty);
172             }
173         }
174         if (matched && NoiseReduction.BAD.contains(this.dirty)) {
175             this.clean.append(
176                 ImmutableList.copyOf(
177                     IOUtils.lineIterator(this.dirty.read(), CharEncoding.UTF_8)
178                 )
179             );
180         }
181         if (NoiseReduction.GOOD.contains(this.dirty)) {
182             this.clean.append(lines);
183         } else {
184             this.dirty.append(lines);
185             NoiseReduction.BAD.add(this.dirty);
186         }
187     }
188 
189     @Override
190     public InputStream read() throws IOException {
191         final boolean exists = Iterables.contains(
192             Iterables.limit(this.dirty.pulses(), this.visible),
193             this.work.scheduled()
194         );
195         InputStream stream;
196         if (exists) {
197             stream = this.dirty.read();
198         } else {
199             stream = this.clean.read();
200         }
201         return new SequenceInputStream(
202             IOUtils.toInputStream(
203                 String.format(
204                     // @checkstyle LineLength (1 line)
205                     "NoiseReduction: exists=%B, pattern='%s', visible=%d\n",
206                     exists, this.pattern, this.visible
207                 )
208             ),
209             stream
210         );
211     }

213     /**
214      * Distinct iterator, not thread-safe.
215      * @see https://code.google.com/p/guava-libraries/issues/detail?id=1464
216      */
217     private static final class Distinct<T> implements Iterator<T> {
218         /**
219          * Original iterator.
220          */
221         private final transient Iterator<T> origin;
222         /**
223          * Set of elements already seen.
224          */
225         private final transient Set<T> seen = new LinkedHashSet<T>(0);
226         /**
227          * Recent element.
228          */
229         private final transient AtomicReference<T> recent =
230             new AtomicReference<T>();
231         /**
232          * Ctor.
233          * @param iterator Original iterator
234          */
235         protected Distinct(final Iterator<T> iterator) {
236             this.origin = iterator;
237         }
238         @Override
239         public boolean hasNext() {
240             while (this.recent.get() == null && this.origin.hasNext()) {
241                 final T next = this.origin.next();
242                 if (!this.seen.contains(next)) {
243                     this.seen.add(next);
244                     this.recent.set(next);
245                 }
246             }
247             return this.recent.get() != null;
248         }
249         @Override
250         public T next() {
251             if (!this.hasNext()) {
252                 throw new NoSuchElementException();
253             }
254             return this.recent.getAndSet(null);
255         }
256         @Override
257         public void remove() {
258             throw new UnsupportedOperationException();
259         }
260     }
