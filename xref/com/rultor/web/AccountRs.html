201     /**
202      * Get entrance page JAX-RS response.
203      * @return The JAX-RS response
204      * @throws IOException If fails on tailing
205      */
206     @GET
207     @Path("/")
208     public Response index() throws IOException {
209         final Sheet sheet = this.sheet();
210         return new PageBuilder()
211             .stylesheet("/xsl/account.xsl")
212             .build(EmptyPage.class)
213             .init(this)
214             .append(new Breadcrumbs().with("account").bundle())
215             .append(new JaxbBundle("sql", sheet.toString()))
216             .append(new JaxbBundle("since", Integer.toString(this.since)))
217             .append(
218                 new JaxbBundle("columns").add(
219                     new JaxbBundle.Group<Column>(sheet.columns()) {
220                         @Override
221                         public JaxbBundle bundle(final Column column) {
222                             return AccountRs.this.column(column);
223                         }
224                     }
225                 )
226             )
227             .link(
228                 new Link(
229                     "latest",
230                     this.home().replaceQueryParam(AccountRs.QUERY_SINCE, 0)
231                 )
232             )
233             .link(new Link("coupon", "./coupon"))
234             .append(this.receipts(sheet.tail(this.since).iterator(), Tv.TWENTY))
235             .render()
236             .build();
237     }

239     /**
240      * Fund account with coupon.
241      * @param code Coupon code to use.
242      * @return The JAX-RS response.
243      */
244     @POST
245     @Path("/coupon")
246     public Response coupon(
247         @NotNull(message = "coupon code is mandatory")
248         @FormParam("code") final String code
249     ) {
250         try {
251             this.user().account().fund(code);
252             throw this.flash().redirect(
253                 this.uriInfo().getBaseUri(),
254                 String.format("Coupon `%s` successfully used", code),
255                 Level.INFO
256             );
257         } catch (final InvalidCouponException ex) {
258             throw this.flash().redirect(
259                 this.uriInfo().getRequestUri(),
260                 String.format(
261                     "Coupon code `%s` is invalid: %s",
262                     code, Exceptions.message(ex)
263                 ),
264                 Level.WARNING
265             );
266         }
267     }

269     /**
270      * Show receipts.
271      * @param receipts Receipts
272      * @param maximum Maximum to show
273      * @return Collection of JAXB rules
274      */
275     private JaxbBundle receipts(final Iterator<List<Object>> receipts,
276         final int maximum) {
277         JaxbBundle bundle = new JaxbBundle("receipts");
278         int pos;
279         for (pos = 0; pos < maximum; ++pos) {
280             if (!receipts.hasNext()) {
281                 break;
282             }
283             bundle = bundle.add(this.receipt(receipts.next()));
284         }
285         if (pos == maximum && receipts.hasNext()) {
286             bundle = bundle.link(
287                 new Link(
288                     "more",
289                     this.home().replaceQueryParam(
290                         AccountRs.QUERY_SINCE, this.since + pos
291                     )
292                 )
293             );
294         }
295         return bundle;
296     }

298     /**
299      * Receipt to bundle.
300      * @param receipt Receipt
301      * @return Bundle
302      */
303     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
304     private JaxbBundle receipt(final List<Object> receipt) {
305         JaxbBundle bundle = new JaxbBundle("receipt");
306         for (int idx = 0; idx < receipt.size(); ++idx) {
307             String cell = receipt.get(idx).toString();
308             if (idx == receipt.size() - 1) {
309                 cell = new Dollars(Long.parseLong(cell)).toString();
310             }
311             bundle = bundle.add("cell", cell).up();
312         }
313         return bundle;
314     }
315 
316     /**
317      * Column to bundle.
318      * @param column Column
319      * @return Bundle
320      */
321     private JaxbBundle column(final Column column) {
322         JaxbBundle bundle = new JaxbBundle("column")
323             .add("title", column.title())
324             .up();
325         if (this.asc.contains(column.title())) {
326             bundle = bundle.attr("sorted", "asc");
327         }
328         if (this.desc.contains(column.title())) {
329             bundle = bundle.attr("sorted", "desc");
330         }
331         if (this.groups.contains(column.title())) {
332             bundle = bundle.attr("grouped", "yes");
333         }
334         if (column.isSum()) {
335             bundle = bundle.attr("sum", "yes");
336         }
337         if (column.isGroup()) {
338             bundle.link(
339                 new Link(
340                     "group",
341                     this.home().queryParam(
342                         AccountRs.QUERY_GROUP, column.title()
343                     )
344                 )
345             );
346         }
347         bundle.link(
348             new Link(
349                 "asc",
350                 this.home().queryParam(AccountRs.QUERY_ASC, column.title())
351             )
352         );
353         bundle.link(
354             new Link(
355                 "desc",
356                 this.home().queryParam(AccountRs.QUERY_DESC, column.title())
357             )
358         );
359         return bundle;
360     }

362     /**
363      * Create URI builder with currently set params.
364      * @return URI builder
365      */
366     private UriBuilder home() {
367         final UriBuilder builder = this.uriInfo()
368             .getBaseUriBuilder()
369             .clone()
370             .path(AccountRs.class)
371             .queryParam(AccountRs.QUERY_SINCE, this.since)
372             .queryParam(AccountRs.QUERY_START, this.start)
373             .queryParam(AccountRs.QUERY_FINISH, this.finish);
374         for (final String grp : this.groups) {
375             builder.queryParam(AccountRs.QUERY_GROUP, grp);
376         }
377         for (final String col : this.asc) {
378             builder.queryParam(AccountRs.QUERY_ASC, col);
379         }
380         for (final String col : this.desc) {
381             builder.queryParam(AccountRs.QUERY_DESC, col);
382         }
383         return builder;
384     }
385 
386     /**
387      * Fetch sheet.
388      * @return Sheet
389      */
390     private Sheet sheet() {
391         Sheet sheet = this.user().account().sheet()
392             .between(this.start, this.finish);
393         for (final String grp : this.groups) {
394             sheet = sheet.groupBy(grp);
395         }
396         for (final String col : this.asc) {
397             sheet = sheet.orderBy(col, true);
398         }
399         for (final String col : this.desc) {
400             sheet = sheet.orderBy(col, false);
401         }
402         return sheet;
403     }
