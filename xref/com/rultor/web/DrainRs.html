64  /**
65   * Drain of a rule.
66   *
67   * @author Yegor Bugayenko (yegor@tpc2.com)
68   * @version $Id$
69   * @since 1.0
70   * @checkstyle MultipleStringLiterals (500 lines)
71   * @checkstyle ClassDataAbstractionCoupling (500 lines)
72   */
73  @Path("/drain/{rule:[\\w\\-]+}")
74  @Loggable(Loggable.DEBUG)
75  @SuppressWarnings("PMD.ExcessiveImports")
76  public final class DrainRs extends BaseRs {
77  
78      /**
79       * Query param.
80       */
81      private static final String QUERY_SINCE = "since";
82  
83      /**
84       * Rule name.
85       */
86      private transient String name;
87  
88      /**
89       * Since (date).
90       */
91      private transient Time since;
92  
93      /**
94       * Inject it from query.
95       * @param rule Rule name
96       */
97      @PathParam("rule")
98      public void setName(@NotNull(message = "rule name can't be NULL")
99          final String rule) {
100         this.name = rule;
101     }
102 
103     /**
104      * Inject it from query.
105      * @param time Since what time
106      */
107     @QueryParam(DrainRs.QUERY_SINCE)
108     public void setSince(final String time) {
109         if (time != null) {
110             this.since = new Time(Long.parseLong(time));
111         }
112     }
113 
114     /**
115      * Get entrance page JAX-RS response.
116      * @return The JAX-RS response
117      */
118     @GET
119     @Path("/")
120     public Response index() {
121         EmptyPage page = new PageBuilder()
122             .stylesheet("/xsl/drain.xsl")
123             .build(EmptyPage.class)
124             .init(this)
125             .link(
126                 new Link(
127                     "edit",
128                     this.uriInfo().getBaseUriBuilder()
129                         .clone()
130                         .path(RuleRs.class)
131                         .build(this.name)
132                 )
133             )
134             .append(
135                 new Breadcrumbs()
136                     .with("rules")
137                     .with("edit", this.name)
138                     .with("self", "drain")
139                     .bundle()
140             )
141             .append(new JaxbBundle("rule", this.name));
142         final Drain drain = this.drain(new Time());
143         Pageable<Time, Time> pulses = this.pulses(drain);
144         final int total;
145         if (this.since == null) {
146             total = Tv.FIVE;
147         } else {
148             try {
149                 pulses = pulses.tail(this.since);
150             } catch (final IOException ex) {
151                 throw this.flash().redirect(
152                     this.uriInfo().getBaseUri(),
153                     String.format(
154                         "I/O problem with the tail of drain of \"%s\": %s",
155                         this.name, Exceptions.message(ex)
156                     ),
157                     Level.SEVERE
158                 );
159             }
160             total = Tv.TWENTY;
161             page = page
162                 .append(new JaxbBundle("since", this.since.toString()))
163                 .link(
164                     new Link(
165                         "latest",
166                         this.uriInfo().getBaseUriBuilder()
167                             .clone()
168                             .path(DrainRs.class)
169                             .build(this.name)
170                     )
171                 );
172         }
173         return page
174             .append(this.pulses(pulses.iterator(), total))
175             .render()
176             .build();
177     }
178 
179     /**
180      * Get drain.
181      * @param time Time where we should create a drain
182      * @return The drain
183      */
184     private Drain drain(final Time time) {
185         final Object src;
186         try {
187             src = this.repo()
188                 .make(this.user(), this.rule().drain())
189                 .instantiate(
190                     this.users(),
191                     new Arguments(
192                         new Coordinates.Simple(
193                             this.user().urn(), this.name, time
194                         ),
195                         new Wallet.Empty()
196                     )
197                 );
198         } catch (final SpecException ex) {
199             throw this.flash().redirect(
200                 this.uriInfo().getBaseUri(),
201                 String.format(
202                     "Can't render drain of \"%s\": %s", this.name,
203                     Exceptions.message(ex)
204                 ),
205                 Level.SEVERE
206             );
207         }
208         return Drain.class.cast(src);
209     }
210 
211     /**
212      * Fetch pulses from drain.
213      * @param drain The drain
214      * @return The pulses
215      */
216     private Pageable<Time, Time> pulses(final Drain drain) {
217         try {
218             return drain.pulses();
219         } catch (final IOException ex) {
220             throw this.flash().redirect(
221                 this.uriInfo().getBaseUri(),
222                 String.format(
223                     "I/O problem with the drain of \"%s\": %s",
224                     this.name,
225                     Exceptions.message(ex)
226                 ),
227                 Level.SEVERE
228             );
229         }
230     }
231 
232     /**
233      * Get rule.
234      * @return The rule
235      */
236     private Rule rule() {
237         try {
238             return this.user().rules().get(this.name);
239         } catch (final NoSuchElementException ex) {
240             throw this.flash().redirect(this.uriInfo().getBaseUri(), ex);
241         }
242     }
243 
244     /**
245      * All pulses of the rule.
246      * @param pulses All pulses to show
247      * @param maximum Maximum to show
248      * @return Collection of JAXB rules
249      */
250     private JaxbBundle pulses(final Iterator<Time> pulses, final int maximum) {
251         JaxbBundle bundle = new JaxbBundle("pulses");
252         int pos;
253         for (pos = 0; pos < maximum; ++pos) {
254             if (!pulses.hasNext()) {
255                 break;
256             }
257             bundle = bundle.add(this.pulse(pulses.next()));
258         }
259         if (pos == maximum && pulses.hasNext()) {
260             bundle = bundle.link(
261                 new Link(
262                     "more",
263                     this.uriInfo()
264                         .getBaseUriBuilder()
265                         .clone()
266                         .path(DrainRs.class)
267                         .queryParam(DrainRs.QUERY_SINCE, pulses.next().millis())
268                         .build(this.name)
269                 )
270             );
271         }
272         return bundle;
273     }
274 
275     /**
276      * Convert pulse to JaxbBundle.
277      * @param time Date of it
278      * @return Bundle
279      */
280     private JaxbBundle pulse(final Time time) {
281         final Collection<Exception> bugs = new LinkedList<Exception>();
282         JaxbBundle bundle = new JaxbBundle("pulse")
283             .add("time", time.toString())
284             .up()
285             .link(
286                 new Link(
287                     // @checkstyle MultipleStringLiterals (1 line)
288                     "stream",
289                     this.uriInfo().getBaseUriBuilder()
290                         .clone()
291                         .path(PulseRs.class)
292                         .path(PulseRs.class, "stream")
293                         .build(this.name, time.millis())
294                 )
295             );
296         final Snapshot snapshot;
297         try {
298             snapshot = new Snapshot(this.drain(time).read());
299             bundle = bundle.add("xembly", snapshot.xembly()).up();
300             try {
301                 bundle = bundle.add(
302                     new XSLT(
303                         snapshot,
304                         this.getClass().getResourceAsStream("post.xsl")
305                     ).dom().getDocumentElement()
306                 );
307             } catch (final XemblyException ex) {
308                 bugs.add(ex);
309             } catch (final TransformerException ex) {
310                 bugs.add(ex);
311             }
312         } catch (final IOException ex) {
313             bugs.add(ex);
314         } catch (final SyntaxException ex) {
315             bugs.add(ex);
316         }
317         return bundle.add(
318             new JaxbBundle("exceptions").add(
319                 new JaxbBundle.Group<Exception>(bugs) {
320                     @Override
321                     public JaxbBundle bundle(final Exception bug) {
322                         return new JaxbBundle("exception")
323                             .add("class", bug.getClass().getCanonicalName())
324                             .up()
325                             .add("message", Exceptions.message(bug))
326                             .up();
327                     }
328                 }
329             )
330         );
331     }
