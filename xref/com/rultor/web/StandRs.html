161     /**
162      * Get entrance page JAX-RS response.
163      * @return The JAX-RS response
164      */
165     @GET
166     @Path("/")
167     public Response index() {
168         EmptyPage page = new PageBuilder()
169             .stylesheet(
170                 this.uriInfo().getBaseUriBuilder()
171                     .clone().path(StylesheetsRs.class).build().toString()
172         )
173             .build(EmptyPage.class)
174             .init(this)
175             .link(
176                 new Link(
177                     "edit",
178                     this.uriInfo().getBaseUriBuilder()
179                         .clone()
180                         .path(AclRs.class)
181                         .build(this.name)
182                 )
183             )
184             .link(
185                 new Link(
186                     "pulse-open",
187                     this.uriInfo().getBaseUriBuilder()
188                         .clone()
189                         .path(StandRs.class)
190                         .queryParam(StandRs.QUERY_OPEN, "")
191                         .build(this.name)
192                 )
193             )
194             .link(
195                 new Link(
196                     "collapse",
197                     this.self(new ArrayList<String>(0))
198                 )
199             );
200         if (this.auth().identity().equals(Identity.ANONYMOUS)) {
201             page = page.append(
202                 new Breadcrumbs()
203                     .with("home")
204                     .with("self", this.name)
205                     .bundle()
206             );
207         } else {
208             page = page.append(
209                 new Breadcrumbs()
210                     .with("stands")
211                     .with("edit", this.name)
212                     .with("collapse", "stand")
213                     .bundle()
214             );
215         }
216         Pulses pulses = this.stand().pulses();
217         for (final String tag : this.tags) {
218             pulses = pulses.query().withTag(tag).fetch();
219         }
220         return page
221             .append(new JaxbBundle("stand", this.name))
222             .append(
223                 new JaxbBundle("filters").add(
224                     new JaxbBundle.Group<String>(this.tags) {
225                         @Override
226                         public JaxbBundle bundle(final String tag) {
227                             return new JaxbBundle("filter", tag);
228                         }
229                     }
230                 )
231             )
232             .append(this.widgets(this.widgets(this.stand().widgets())))
233             .append(this.pulses(pulses.iterator(), Tv.TWENTY))
234             .render()
235             .build();
236     }

238     /**
239      * Get snapshot HTML for a pulse.
240      * @param uid Unique identifier of a pulse
241      * @return The JAX-RS response
242      */
243     @GET
244     @Path("/fetch")
245     @Produces(MediaType.TEXT_HTML)
246     public Response fetch(@QueryParam(StandRs.QUERY_ID) final String uid) {
247         final Coordinates coords = Coordinates.Simple.valueOf(uid);
248         Response resp;
249         try {
250             resp = Response.ok().entity(
251                 new XSLT(
252                     this.render(
253                         new JaxbBundle("div"),
254                         this.stand().pulses().tail(coords).iterator().next()
255                     ).element(),
256                     this.getClass().getResourceAsStream("fetch.xsl")
257                 ).xml()
258             ).build();
259         } catch (final TransformerException ex) {
260             resp = Response.serverError().entity(
261                 Exceptions.stacktrace(ex)
262             ).build();
263         } catch (final IOException ex) {
264             resp = Response.serverError().entity(
265                 Exceptions.stacktrace(ex)
266             ).build();
267         }
268         return resp;
269     }

271     /**
272      * Get stand.
273      * @return The stand
274      */
275     private Stand stand() {
276         final Stand stand;
277         try {
278             stand = this.users().stand(this.name);
279         } catch (final NoSuchElementException ex) {
280             throw this.flash().redirect(this.uriInfo().getBaseUri(), ex);
281         }
282         if (!stand.owner().equals(this.auth().identity().urn())
283             && !this.acl(stand).canView(this.auth().identity().urn())) {
284             throw this.flash().redirect(
285                 this.uriInfo().getBaseUri(),
286                 String.format("access denied to stand `%s`", this.name),
287                 Level.WARNING
288             );
289         }
290         return stand;
291     }

293     /**
294      * All pulses of the stand.
295      * @param pulses All pulses to show
296      * @param maximum Maximum to show
297      * @return Collection of JAXB stands
298      */
299     private JaxbBundle pulses(final Iterator<Pulse> pulses, final int maximum) {
300         JaxbBundle bundle = new JaxbBundle("pulses");
301         int pos;
302         for (pos = 0; pos < maximum; ++pos) {
303             if (!pulses.hasNext()) {
304                 break;
305             }
306             bundle = bundle.add(this.pulse(pulses.next()));
307         }
308         return bundle;
309     }

311     /**
312      * Convert pulse to JaxbBundle.
313      * @param pulse The pulse
314      * @return Bundle
315      */
316     private JaxbBundle pulse(final Pulse pulse) {
317         final Coordinates coords = pulse.coordinates();
318         JaxbBundle bundle = new JaxbBundle("pulse")
319             .add("coordinates")
320             .add("rule", coords.rule()).up()
321             .add("owner", coords.owner().toString()).up()
322             .add("scheduled", coords.scheduled().toString()).up()
323             .up();
324         final String label = new Coordinates.Simple(coords).toString();
325         final ArraySet<String> now = new ArraySet<String>(this.open);
326         if (this.open.contains(label)) {
327             bundle = this
328                 .render(bundle, pulse)
329                 .link(new Link("close", this.self(now.without(label))))
330                 .link(
331                     new Link(
332                         "fetch",
333                         this.uriInfo().getBaseUriBuilder()
334                             .clone()
335                             .path(StandRs.class)
336                             .path(StandRs.class, "fetch")
337                             .queryParam(StandRs.QUERY_ID, "{id}")
338                             .build(this.name, label)
339                     )
340                 );
341         } else {
342             bundle = bundle
343                 .link(new Link("open", this.self(now.with(label))))
344                 .add("tags")
345                 .add(
346                     new JaxbBundle.Group<Tag>(pulse.tags()) {
347                         @Override
348                         public JaxbBundle bundle(final Tag tag) {
349                             return StandRs.this.bundle(tag);
350                         }
351                     }
352                 )
353                 .up();
354         }
355         return bundle;
356     }

358     /**
359      * Build URI to itself with new list of open pulses.
360      * @param coords Names of pulses
361      * @return URI
362      */
363     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
364     private URI self(final Collection<String> coords) {
365         final Object[] args = new Object[coords.size() + 1];
366         final Object[] labels = new String[coords.size()];
367         args[0] = this.name;
368         int idx = 0;
369         for (final String coord : coords) {
370             labels[idx] = String.format("{arg%d}", idx);
371             args[idx + 1] = coord;
372             ++idx;
373         }
374         return this.uriInfo().getBaseUriBuilder()
375             .clone()
376             .path(StandRs.class)
377             .queryParam(StandRs.QUERY_OPEN, labels)
378             .build(args);
379     }

381     /**
382      * Render snapshot into bundle.
383      * @param bundle Bundle to render into
384      * @param pulse The pulse
385      * @return Bundle
386      */
387     private JaxbBundle render(final JaxbBundle bundle, final Pulse pulse) {
388         JaxbBundle output = bundle;
389         try {
390             final Snapshot snapshot = this.snapshot(pulse.xembly());
391             try {
392                 output = output.add(
393                     new XSLT(
394                         snapshot,
395                         this.getClass().getResourceAsStream("post.xsl")
396                     ).dom().getDocumentElement()
397                 );
398             } catch (final XemblyException ex) {
399                 output = this.bug(output, ex);
400             } catch (final TransformerException ex) {
401                 output = this.bug(output, ex);
402             }
403         } catch (final IOException ex) {
404             output = this.bug(output, ex);
405         } catch (final SyntaxException ex) {
406             output = this.bug(output, ex);
407         }
408         return output;
409     }

411     /**
412      * Get snapshot from xembly.
413      * @param xembly Xembly script
414      * @return Its snapshot
415      * @throws SyntaxException If fails
416      */
417     private Snapshot snapshot(final String xembly) throws SyntaxException {
418         return new Snapshot(
419             new Directives(xembly).xpath("/snapshot/spec").remove()
420         );
421     }

433     /**
434      * All widgets of the stand.
435      * @param spec Spec of array of widgets
436      * @return Collection of JAXB widgets
437      */
438     @SuppressWarnings("unchecked")
439     private Collection<Widget> widgets(final Spec spec) {
440         Collection<Widget> list;
441         try {
442             list = Collection.class.cast(
443                 this.repo().make(new User.Nobody(), spec).instantiate(
444                     this.users(),
445                     new Arguments(
446                         new Coordinates.None(), new Wallet.Empty()
447                     )
448                 )
449             );
450         } catch (final SpecException ex) {
451             list = Arrays.<Widget>asList(new Alert(Exceptions.stacktrace(ex)));
452         }
453         return list;
454     }

456     /**
457      * All widgets of the stand.
458      * @param widgets Collection of widgets
459      * @return Collection of JAXB widgets
460      */
461     private JaxbBundle widgets(final Collection<Widget> widgets) {
462         JaxbBundle bundle = new JaxbBundle("widgets");
463         for (final Widget widget : widgets) {
464             bundle = bundle.add(this.widget(widget));
465         }
466         return bundle;
467     }

469     /**
470      * Render one widget.
471      * @param widget Widget to render
472      * @return DOM element
473      */
474     private Element widget(final Widget widget) {
475         final Document dom;
476         try {
477             dom = DocumentBuilderFactory.newInstance()
478                 .newDocumentBuilder().newDocument();
479         } catch (final ParserConfigurationException ex) {
480             throw new IllegalStateException(ex);
481         }
482         final Element root = dom.createElement("widget");
483         root.setAttribute("class", widget.getClass().getCanonicalName());
484         dom.appendChild(root);
485         try {
486             new Xembler(
487                 new Directives()
488                     .xpath("/widget")
489                     .append(widget.render(this.stand()))
490             ).apply(dom);
491         } catch (final ImpossibleModificationException ex) {
492             final Element error = dom.createElement("error");
493             error.setTextContent(Exceptions.stacktrace(ex));
494             dom.getDocumentElement().appendChild(error);
495         }
496         return dom.getDocumentElement();
497     }

499     /**
500      * Bundle a tag.
501      * @param tag A tag
502      * @return Bundle
503      */
504     private JaxbBundle bundle(final Tag tag) {
505         final Set<String> labels = new TreeSet<String>();
506         labels.addAll(this.tags);
507         labels.add(tag.label());
508         final Object[] args = new String[labels.size()];
509         final Object[] values = new String[labels.size() + 1];
510         int idx = 0;
511         values[0] = this.name;
512         for (final String label : labels) {
513             args[idx] = String.format("{arg%d}", idx);
514             values[idx + 1] = label;
515             ++idx;
516         }
517         return new JaxbBundle("tag")
518             .add("label", tag.label())
519             .up()
520             .add("level", tag.level().toString())
521             .up()
522             .add("markdown", tag.markdown())
523             .up()
524             .link(
525                 new Link(
526                     "filter",
527                     this.uriInfo().getBaseUriBuilder()
528                         .clone()
529                         .path(StandRs.class)
530                         .queryParam(StandRs.QUERY_TAGS, args)
531                         .build(values)
532                 )
533             );
534     }
